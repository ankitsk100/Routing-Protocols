enable
configure terminal
hostname R4
no ip domain-lookup

interface g0/0
 description R4 LAN
 ip address 192.168.4.1 255.255.255.0
 no shutdown

interface g0/1
 description Link to R2 (Area 20)
 ip address 10.0.24.2 255.255.255.252
 no shutdown

router ospf 10
 router-id 4.4.4.4
 passive-interface g0/0
 network 192.168.4.0 0.0.0.255 area 20
 network 10.0.24.0 0.0.0.3 area 20

end
write memory
